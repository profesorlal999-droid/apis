Intelligent sketch completion. Draws standard icons based on your rough strokes.

Step 1: Predict POST /api/run/autodraw/predict

Step 2: Get Icon GET /api/run/autodraw/icon

Interactive Example

Copy this code to implement the full AutoDraw flow:

<!-- HTML -->
<div style="display:flex; gap:10px;">
    <canvas id="adCanvas" width="400" height="300" style="border:1px solid #333; background:#fff; cursor:crosshair;"></canvas>
    <div id="ad-buttons" style="display:flex; flex-direction:column; gap:5px; max-height:300px; overflow-y:auto; width:150px;">
        <small style="color:#666">Draw something...</small>
    </div>
</div>
<button onclick="predictAutoDraw()" style="margin-top:10px;">PREDICT</button>
<button onclick="clearCanvas()" style="margin-top:10px;">CLEAR</button>

<!-- SCRIPT -->
<script>
const adCanvas = document.getElementById('adCanvas');
const adCtx = adCanvas.getContext('2d');
let adIsDrawing = false;
let adInk = []; 
let adStroke = [[], [], []];
let adStartTime = Date.now();
// REPLACE THIS WITH YOUR REAL API KEY
const API_KEY = "sk-nx-XXXXXXXXXXXXXXXX"; 

// 1. Drawing Logic
adCanvas.addEventListener('mousedown', (e) => {
    adIsDrawing = true;
    adStroke = [[], [], []];
    adCtx.beginPath();
    adCtx.lineWidth = 3;
    adCtx.moveTo(e.offsetX, e.offsetY);
});

adCanvas.addEventListener('mousemove', (e) => {
    if (!adIsDrawing) return;
    adCtx.lineTo(e.offsetX, e.offsetY);
    adCtx.stroke();
    adStroke[0].push(e.offsetX);
    adStroke[1].push(e.offsetY);
    adStroke[2].push(Date.now() - adStartTime);
});

adCanvas.addEventListener('mouseup', () => {
    adIsDrawing = false;
    if (adStroke[0].length > 0) adInk.push(adStroke);
});

function clearCanvas() {
    adCtx.clearRect(0, 0, adCanvas.width, adCanvas.height);
    adInk = [];
    document.getElementById('ad-buttons').innerHTML = '<small style="color:#666">Draw something...</small>';
}

// 2. Get Suggestions
async function predictAutoDraw() {
    const btnContainer = document.getElementById('ad-buttons');
    btnContainer.innerHTML = 'Loading...';

    const response = await fetch('https://nexus-api-2026.vercel.app/api/run/autodraw/predict', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
            key: API_KEY,
            ink: adInk,
            width: adCanvas.width,
            height: adCanvas.height
        })
    });

    const data = await response.json();
    btnContainer.innerHTML = '';

    if(data.suggestions) {
        data.suggestions.forEach(name => {
            const btn = document.createElement('button');
            btn.innerText = name;
            btn.style.cssText = "padding:5px; background:#222; color:#fff; border:1px solid #444; cursor:pointer;";
            // When clicked, fetch and draw SVG
            btn.onclick = () => fetchAndDrawIcon(name);
            btnContainer.appendChild(btn);
        });
    }
}

// 3. Fetch SVG and Replace Canvas
async function fetchAndDrawIcon(name) {
    // Calling backend to proxy the Google Storage request
    const url = `https://nexus-api-2026.vercel.app/api/run/autodraw/icon?key=${API_KEY}&name=${name}&index=0`;
    
    const response = await fetch(url);
    if(response.ok) {
        const svgText = await response.text();
        
        // Render SVG to Canvas
        const img = new Image();
        const svgBlob = new Blob([svgText], {type: 'image/svg+xml;charset=utf-8'});
        const urlObj = URL.createObjectURL(svgBlob);
        
        img.onload = () => {
            // Clear and draw centered
            adCtx.clearRect(0, 0, adCanvas.width, adCanvas.height);
            adCtx.drawImage(img, 0, 0, adCanvas.width, adCanvas.height);
            URL.revokeObjectURL(urlObj);
        };
        img.src = urlObj;
    } else {
        alert("Icon not found for this variation.");
    }
}
</script>

async function generateImage() {
    const url = "https://nexus-api-2026.vercel.app/api/run/image";
    
    // POST request allows longer prompts
    const response = await fetch(url, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
            key: "sk-nx-XXXXXXXXXXXXXXXX",
            prompt: "A cute robot eating a banana"
        })
    });

    if (response.ok) {
        const imageBlob = await response.blob();
        const imageUrl = URL.createObjectURL(imageBlob);
        const img = document.createElement('img');
        img.src = imageUrl;
        document.body.appendChild(img);
    }
}

Access the advanced coding Nexus agent. Optimized for complex tasks and coding.

Endpoint: /api/run/agent

Method: GET or POST

Cost: 1500 Tokens / Request

Python (POST Streaming)

Reads the response token by token. Best for long inputs.

import requests

API_URL = "https://nexus-api-2026.vercel.app/api/run/agent"
payload = {
    "key": "sk-nx-XXXXXXXXXXXXXXXX",
    "prompt": "Write a python script for snake game",
    "stream": True  # Enable real-time streaming
}

print("Agent thinking...", end="", flush=True)

with requests.post(API_URL, json=payload, stream=True) as r:
    for chunk in r.iter_content(chunk_size=None):
        if chunk:
            print(chunk.decode('utf-8'), end="", flush=True)

Modern browser implementation using POST.

async function askAgent() {
    const url = "https://nexus-api-2026.vercel.app/api/run/agent";
    
    const response = await fetch(url, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
            key: "sk-nx-XXXXXXXXXXXXXXXX",
            prompt: "Explain quantum computing",
            stream: true
        })
    });
    
    const reader = response.body.getReader();
    const decoder = new TextDecoder();

    while (true) {
        const { done, value } = await reader.read();
        if (done) break;
        const chunk = decoder.decode(value);
        console.log(chunk); 
    }
}
